MODULE Pattern13;
  TYPE
    P0 = POINTER TO R0;
    P1 = POINTER TO R1;
    P2 = POINTER TO R2;
    R0 = RECORD x: INTEGER END ;
    R1 = RECORD (R0) y: INTEGER END ;
    R2 = RECORD (R1) z: INTEGER END ;
  VAR
    p0: P0;
    p1: P1;
    p2: P2;
BEGIN
  p0.x := 0;
  p1.y := 1;
  p0(P1).y := 3;
  p0(P2).z := 5;
  IF p1 IS P2 THEN
    p0 := p2
  END
END Pattern13.
